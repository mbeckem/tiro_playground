713  template<typename Callback>
714  struct callback_wrapper {
715  Callback cb;
716 
717  template<typename T>
718  callback_wrapper(in_place_t, T&& t)
719  : cb(std::forward<T>(t)) {}
720 
721  callback_wrapper(callback_wrapper&&) noexcept = delete;
722  callback_wrapper& operator=(callback_wrapper&&) noexcept = delete;
723 
724  static void invoke(tiro_vm_t raw_vm, tiro_handle_t raw_coroutine, void* userdata) {
725  try {
726  vm& v = vm::unsafe_from_raw_vm(raw_vm);
727  coroutine coro = make_copy(v, raw_coroutine).as<coroutine>();
728  static_cast<callback_wrapper*>(userdata)->cb(v, coro);
729  } catch (...) {
730  // Cannot throw across c call boundary.
731  // TODO: Implement exception mechanism
732  std::terminate();
733  }
734  }
735 
736  static void cleanup(void* userdata) { delete static_cast<callback_wrapper*>(userdata); }
737  };
